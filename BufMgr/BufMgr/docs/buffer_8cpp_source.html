85  void BufMgr::allocBuf(FrameId & frame)
86  {
87  while (true) {
88  for (int i = 0; i != (signed)numBufs; i++) {
89  advanceClock();
90  //当某个帧的valid位为false时，说明这个页面不可用，
91  //它就可以被清理掉从而腾出需要的空闲帧，函数返回
92  if (!bufDescTable[clockHand].valid) {
93  frame = clockHand;
94  return;
95  }
96  //当某个帧的refbit位为true时，说明这个页面最近被使用过并且未被替换，
97  //因此该位置不是空闲帧，进入下一次循环
98  if (bufDescTable[clockHand].refbit) {
99  bufDescTable[clockHand].refbit = false;
100  continue;
101  }
102  //当某个帧的pinCnt>0时，说明这个页面正在被使用，不是空闲帧
103  if (bufDescTable[clockHand].pinCnt > 0) {
104  continue;
105  }
106  //当某个帧的dirty位为true时，说明这个页面是脏的，应当将该页面写回磁盘
107  if (bufDescTable[clockHand].dirty) {
108  bufDescTable[clockHand].file->writePage(bufPool[clockHand]);
109  }
110  try {
111  if (bufDescTable[clockHand].file) {
112  hashTable->remove(bufDescTable[clockHand].file,
113  bufDescTable[clockHand].pageNo);
114  bufDescTable[clockHand].Clear();
115  }
116  }catch (HashNotFoundException &e) {
117  }
118  frame = clockHand;
119  return;
120  }
121 
122  //当所有的页面都被占用时，抛出异常
123  bool allPinned = false;
124  for (int i = 0; i < (signed)numBufs; i++) {
125  if (!bufDescTable[i].pinCnt) {
126  allPinned = true;
127  break;
128  }
129  }
130  if (!allPinned) {
131  throw BufferExceededException();
132  }
133  }

146  void BufMgr::readPage(File* file, const PageId pageNo, Page*& page)
147  {
148  FrameId id;
149  try {
150  // Page is in the buffer pool
151  hashTable->lookup(file, pageNo, id);
152  bufDescTable[id].pinCnt++;
153  }
154  catch (HashNotFoundException e) {
155  // Page is not in the buffer pool.
156  // Allocate a buffer frame. Read the page from disk
157  // Insert the page into the hashtable. Set the frame
158  Page pageTemp = file->readPage(pageNo);
159  this->allocBuf(id);
160  bufPool[id] = pageTemp;
161  hashTable->insert(file, pageNo, id);
162  bufDescTable[id].Set(file, pageNo);
163  }
164  bufDescTable[id].refbit = true;
165  // Return a pointer to the frame containing the page
166  page = &bufPool[id];
